// Global constant list of available slots
APPOINTMENT_SLOTS = ["09:00 AM", "10:30 AM", "01:00 PM", "03:30 PM"]

BOOKED_APPOINTMENTS = []

//--------------------------------------------------
// Method to display available appointments
//--------------------------------------------------
method displayAvailableAppointments(SERVICE_TYPE, DATE)

    BOOKED_APPOINTMENTS = DATABASE.getBookedAppointments(SERVICE_TYPE, DATE)

    loop I from 0 to LENGTH(APPOINTMENT_SLOTS) - 1
        if APPOINTMENT_SLOTS[I] NOT IN BOOKED_APPOINTMENTS then
            output APPOINTMENT_SLOTS[I] & " (Available)"
            output "Book this slot? (yes/no)"
            input USER_SELECTION

            if USER_SELECTION = "yes" then
                call bookService(SERVICE_TYPE, DATE, APPOINTMENT_SLOTS[I])
            end if
        else
            output APPOINTMENT_SLOTS[I] & " (Unavailable)"
        end if
    end loop

end method

//--------------------------------------------------
// Method to book an appointment
//--------------------------------------------------
method bookService(SERVICE_TYPE, DATE, TIME_SLOT)

    output "Select Language:"
    input LANGUAGE

    output "Any special requests?"
    input SPECIAL_REQUEST

    if TIME_SLOT NOT IN BOOKED_APPOINTMENTS then

        // Insert booking record into the database
        DATABASE.insertBooking(SERVICE_TYPE, DATE, TIME_SLOT, LANGUAGE, SPECIAL_REQUEST, "pending")

        // Update the booked appointments
        BOOKED_APPOINTMENTS.addItem(TIME_SLOT)
        DATABASE.saveBookedAppointments(SERVICE_TYPE, DATE, BOOKED_APPOINTMENTS)

        output "Booking confirmed for " & TIME_SLOT
        EMAIL.sendConfirmation(SERVICE_TYPE, DATE, TIME_SLOT)

    else
        output "Slot Unavailable"
    end if

end method

input BOOKING_ID

BOOKING = DATABASE.findBooking(BOOKING_ID)

if BOOKING.belongsTo(USER) = true then

    output "Choose Action: "
    output "1. Cancel"
    output "2. Update"
    input ACTION

    if ACTION = 1 then
        DATABASE.deleteBooking(BOOKING_ID)
        EMAIL.sendCancellation(USER, BOOKING_ID)
        output "Booking Cancelled"

    else if ACTION = 2 then
        output "Enter new Date and Time:"
        input NEW_DATE, NEW_TIME

        output "Enter new Service Type:"
        input NEW_SERVICE_TYPE

        output "Enter new Language:"
        input NEW_LANGUAGE

        output "Enter new Special Request:"
        input NEW_SPECIAL_REQUEST

        // Update booking details
        BOOKING.update(NEW_DATE, NEW_TIME, NEW_SERVICE_TYPE, NEW_LANGUAGE, NEW_SPECIAL_REQUEST)

        // Validate updated data
        VALID = DATABASE.validateBooking(BOOKING)

        if VALID = true then
            DATABASE.saveBooking(BOOKING)
            EMAIL.sendUpdateConfirmation(USER, BOOKING_ID)
            output "Booking Updated"
        else
            output "Error: Invalid data. Please try again."
        end if
    end if

else
    output "You are not authorized to change this booking"
end if

END
